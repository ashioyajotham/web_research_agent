import pandas as pd

def analyze_epoch_ai_dataset(filepath):
    """
    Analyzes the structure of an Epoch AI dataset to identify relevant columns.

    Args:
        filepath (str): The path to the downloaded Epoch AI dataset (CSV or similar).

    Returns:
        tuple: A tuple containing four strings representing the column names for 
               model name, compute used, training completion timestamp, and compute unit, 
               or None if any of the columns are not found.  Raises exceptions for file errors.

    Raises:
        FileNotFoundError: If the provided filepath does not exist.
        pd.errors.EmptyDataError: if the file is empty.
        pd.errors.ParserError: if there's an issue parsing the file.
        ValueError: if any of the expected columns are not found.

    """
    try:
        df = pd.read_csv(filepath)  #Efficiently reads the file using pandas
        if df.empty:
            raise pd.errors.EmptyDataError("The dataset file is empty.")

        #Robust column name identification, handling variations in capitalization and spacing
        model_name_col = next((col for col in df.columns if 'model' in col.lower() and 'name' in col.lower()), None)
        compute_used_col = next((col for col in df.columns if 'compute' in col.lower() and ('used' in col.lower() or 'cost' in col.lower())), None)
        timestamp_col = next((col for col in df.columns if 'timestamp' in col.lower() or 'completion' in col.lower() or 'finished' in col.lower()), None)
        compute_unit_col = next((col for col in df.columns if 'unit' in col.lower() or 'flops' in col.lower() or 'tpu' in col.lower()), None)


        if not all([model_name_col, compute_used_col, timestamp_col, compute_unit_col]):
            raise ValueError("Could not identify all necessary columns in the dataset.")

        return model_name_col, compute_used_col, timestamp_col, compute_unit_col

    except FileNotFoundError:
        raise FileNotFoundError(f"Error: File not found at path: {filepath}")
    except (pd.errors.EmptyDataError, pd.errors.ParserError) as e:
        raise type(e)(f"Error reading or parsing the dataset file: {e}")
    except StopIteration:
      raise ValueError("Could not identify all necessary columns in the dataset.")

from typing import List, Dict, Any

def extract_training_runs(dataset: List[List[Any]], 
                          date_col: str, 
                          model_name_col: str, 
                          max_compute_col: str, 
                          compute_unit_col: str) -> List[Dict[str, Any]]:
    """
    Extracts training run data into a list of dictionaries.

    Args:
        dataset: A list of lists representing the dataset.  Each inner list is a row.
        date_col: The name of the column containing dates.
        model_name_col: The name of the column containing model names.
        max_compute_col: The name of the column containing maximum compute used.
        compute_unit_col: The name of the column containing compute units.

    Returns:
        A list of dictionaries, where each dictionary represents a training run 
        with keys: 'Date', 'Model Name', 'Maximum Compute Used', 'Compute Unit'.
        Returns an empty list if the input dataset is empty or invalid.

    Raises:
        ValueError: If any of the specified column names are not found in the dataset
                     or if the dataset is improperly formatted.
        TypeError: if input dataset is not a list of lists.

    """
    if not isinstance(dataset, list):
        raise TypeError("Dataset must be a list of lists.")
    if not dataset:
        return []
    
    header = dataset[0]
    data = dataset[1:]

    try:
        date_index = header.index(date_col)
        model_name_index = header.index(model_name_col)
        max_compute_index = header.index(max_compute_col)
        compute_unit_index = header.index(compute_unit_col)
    except ValueError as e:
        raise ValueError(f"Column name not found in dataset: {e}") from None

    training_runs = []
    for row in data:
        if len(row) != len(header):
            raise ValueError("Inconsistent number of columns in dataset.")
        try:
            training_runs.append({
                'Date': row[date_index],
                'Model Name': row[model_name_index],
                'Maximum Compute Used': row[max_compute_index],
                'Compute Unit': row[compute_unit_index]
            })
        except IndexError as e:
            raise ValueError("Improperly formatted dataset row.") from e

    return training_runs

from datetime import datetime

def filter_max_compute(data):
    """
    Sorts a list of dictionaries by 'Date', then iterates to find entries representing new maximum compute used.

    Args:
        data: A list of dictionaries, where each dictionary must contain 'Date' (datetime object) and 'ComputeUsed' (numeric).

    Returns:
        A list of dictionaries representing new records for maximum compute.  Returns an empty list if input is invalid.

    Raises:
        TypeError: if input data is not a list.
        ValueError: if dictionaries are missing required keys or have incorrect data types.

    """
    if not isinstance(data, list):
        raise TypeError("Input data must be a list.")

    if not all(isinstance(item, dict) for item in data):
        raise TypeError("Input list must contain only dictionaries.")

    for item in data:
        if not all(key in item for key in ['Date', 'ComputeUsed']):
            raise ValueError("Dictionaries must contain 'Date' and 'ComputeUsed' keys.")
        try:
            datetime.fromisoformat(item['Date'])
            float(item['ComputeUsed'])
        except (ValueError, TypeError):
            raise ValueError("'Date' must be a valid datetime string and 'ComputeUsed' must be numeric.")


    data.sort(key=lambda x: x['Date'])  #Sort in place to avoid creating a copy.

    max_compute = float('-inf')
    result = []
    for item in data:
        if item['ComputeUsed'] > max_compute:
            max_compute = item['ComputeUsed']
            result.append(item)
    return result
